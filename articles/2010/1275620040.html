Extract privatekey and publickey from p12

$ openssl pkcs12 -in seanlee_yessign.p12 -clcerts -nokeys -out cert.pem
$ openssl x509 -in cert.pem -pubkey -noout > pub.key

$ cat pub.key 
$ openssl rsa -text -pubin -in pub.key -noout
